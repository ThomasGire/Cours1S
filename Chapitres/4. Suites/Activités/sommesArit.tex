\begin{enumerate}

 \item Calculer $\sum\limits_{i=1}^{13} i=1+2+3+...+13$.
 \item Calculer $\sum\limits_{i=1}^{30} i=1+2+3+...+30$.
 \item Donner une définition par récurence de la suite définie pour tout entier $n \geq 1$ par
 $S_n=\sum\limits_{i=1}^{n} i$.
 \item Conjecturer une formule explicite pour $S_n$.
 \item Démontrer cette conjecture.
 \item Calculer $S_{63}$.
 \item Calculer $\sum\limits_{i=14}^{30} i=14+15+...+30$.
 \item Trouver une formule pour 
 $\sum\limits_{i=p}^{n} i=p+(p+1)+...+(n-1)+n$.
 \item Soit $(u_n)_{n \geq 0}$ la suite arithmétique
 de raison 2 telle que $u_5=1$.
 
 Calculer $\sum\limits_{i=5}^{21} u_i=u_5+u_6+...+u_{20}+u_{21}$.
 \item Donner une formule pour 
 $\sum\limits_{i=p}^{n} u_i=u_p+u_{p+1}+...+u_{n-1}+u_{n}$
  dans le cas où la suite $(u_n)$ est arithmétique.
 
 
\end{enumerate}
