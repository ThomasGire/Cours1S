\vskip\parskip

\hglue\leftmarginii$\coord{\Vec{AF}}{\Vec{AD}}$%
\quad\string;\quad%
$\coord{\Vec{DF}}{\Vec{DA}}$

\item En d\'eduire la mesure de l'angle orient\'e $\coord{\Vec{DC}}{\Vec{DE}}$.
\end{enumerate}

\item En d\'eduire que les points $D$, $F$ et $E$ sont align\'es.
\end{enumerate}
